-- WAQTD DNO AND NUMBER OF EMP WORKING IN EACH DEPT
-- IF THERE ARE AT LEAST 2 CLERKS IN EACH DEPT
SELECT DEPTNO, COUNT(*) AS NO_OF_CLERKS
FROM EMP
WHERE JOB = 'CLERK'
GROUP BY DEPTNO
HAVING COUNT(*) >= 2;

-- IF THERE ARE AT LEAST 4 EMP IN EACH DEPT
SELECT DEPTNO, SUM(SAL) AS TOTAL_SAL
FROM EMP

-- WAQTD NUMBER OF EMP EARNING SAL MORE THAN 1200 IN EACH JOB
-- AND THE TOTAL SAL NEEDED TO PAY EMP OF EACH JOB MUST EXCEED 3800
SELECT JOB, COUNT(*) AS NO_EMP
FROM EMP
WHERE SAL > 1200
GROUP BY JOB
HAVING SUM(SAL) > 3800;

-- WAQTD DEPTNO IF THERE ARE AT LEAST 3 EMP IN EACH DEPT
-- WHOSE NAME HAS CHAR 'A' OR 'S'
SELECT DEPTNO
FROM EMP
WHERE ENAME LIKE '%A%' OR ENAME LIKE '%S%'
GROUP BY DEPTNO
HAVING COUNT(*) >= 3;

-- WAQTD MIN AND MAX SALARIES OF EACH JOB
-- IF MIN SAL IS MORE THAN 1000 AND MAX SAL IS LESS THAN 5000
SELECT JOB, MIN(SAL) AS MIN, MAX(SAL) AS MAX
FROM EMP
GROUP BY JOB
HAVING MIN(SAL) > 1000
AND MAX(SAL) < 5000;
